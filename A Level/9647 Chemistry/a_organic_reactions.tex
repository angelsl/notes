\chapter{Organic Chemistry Summary of Reactions}
\section{Free Radical Substitution}
\begin{enumerate}
    \item \ch{R-CH3 + X2 -> R-CH2X + HX}; reagents: \ch{X2~(g)}; conditions: r.t.p., UV light
\end{enumerate}
\section{Electrophilic Addition}
\begin{enumerate}
    \item \ch{CH2=CH2 + X2 -> CH2XCH2X}; reagents: \ch{X2~(CCl4)}; conditions: r.t.p., dark; observations: decolourisation of reddish-brown \ch{Br2} or greenish-yellow \ch{Cl2} solution
    \item \ch{CH2=CH2 + HX -> CH3CH2X}; reagents: dry \ch{HX~(g)} or \ch{HX~(CCl4)}; conditions: r.t.p.
    \item \ch{CH2=CH2 + H2O -> CH3CH2OH}; reagents: \ch{H2O~(g)} or \ch{H2O~(l)}; conditions: \SI{300}{\celsius}, \SI{65}{\atmosphere}, conc. \ch{H3PO4} cat. \textbf{or} cold conc. \ch{H2SO4} then hot water
\end{enumerate}
\section{Electrophilic Substitution}
\begin{enumerate}
    \item \ch{C6H6 + HNO3 -> C6H5NO2 + H2O}; reagents: conc. \ch{HNO3}, conc. \ch{H2SO4} cat.; conditions: \SI{55}{\celsius}
    \item \ch{C6H6 + X2 -> C6H5X + HX}; reagents: \ch{X2}; conditions: r.t.p., dark, ahd. \ch{FeX3} or \ch{AlX3} or \ch{Fe} cat.; observations: white fumes of \ch{HX}
    \item \ch{C6H6 + RX -> C6H5R + HX}; reagents: \ch{RX}; conditions: r.t.p., ahd. \ch{FeX3} or \ch{AlX3} cat.
    \item \ch{C6H5OH + HNO3 -> "o-/p-" C6H4(OH)(NO2) + H2O}; reagents: dil. \ch{HNO3}; conditions: r.t.p. (heat for disub.)
    \item \ch{C6H5OH + 3 HNO3 -> C6H2(OH)(NO2)3 + 3 H2O}; reagents: conc. \ch{HNO3}; conditions: r.t.p.
    \item \ch{C6H5OH + X2 -> "o-/p-" C6H4(OH)(X) + HX}; reagents: \ch{X2~(CCl4)}; conditions: r.t.p.
    \item \ch{C6H5OH + 3 X2 -> C6H2(OH)(X)3 + 3 HX}; reagents: \ch{X2~(aq)}; conditions: r.t.p. Test for phenol with \ch{Br2}; observations: white ppt of 2,4,6-tribromophenol
    \item \ch{C6H5NH2 + 3 X2 -> C6H2(NH2)(X)3 + 3 HX}; reagents: \ch{X2~(aq)}; conditions: r.t.p. Test for phenylamine with \ch{Br2}; observations: white ppt of 2,4,6-tribromophenylamine. Note: same observations as phenol!
\end{enumerate}
\section{Nucleophilic Substitution}
\begin{enumerate}
    \item \ch{RX + NaOH -> ROH + NaX}; reagents: \ch{NaOH~(aq)} or \ch{KOH~(aq)}; conditions: heat. Test for \ch{RX}; observations: ppt (white: \ch{RCl}; cream: \ch{RBr}; yellow: \ch{RI}) forms with \ch{AgNO3~(aq)}
    \item \ch{RX + KCN -> RCN + KX}; reagents: \ch{KCN} in ethanol; conditions: heat
    \item \ch{RX + 2 NH3 -> RNH2 + NH4X}; reagents: exc. conc. \ch{NH3} in ethanol; conditions: heat in sealed tube
    \item \ch{RCOOH + R'OH <=> RCOOR' + H2O}; reagents: \ch{RCOOH} or \ch{R'OH} (Nu: \ch{R'O-}); conditions: conc. \ch{H2SO4} cat., heat
    Note: \ch{R'} cannot be benzene in this case (phenol will not undergo N.S. with \ch{-COOH}.)
    \item \ch{RCOCl + R'OH -> RCOOR' + HCl}; reagents: \ch{RCOCl} or \ch{R'OH} (Nu: \ch{R'O-}); conditions: r.t.p.
    \item \ch{RCOCl + NHR'R'' -> RCONR'R'' + HCl}; reagents: \ch{NHR'R''}; conditions: r.t.p.
    \item \ch{3 ROH + PX3 -> 3 RX + H3PO3}; reagents: dry \ch{PX3} \textbf{or} \ch{X2} with red phosphorus; conditions: heat
    \item \ch{ROH + HCl -> RCl + H2O}; reagents: \ch{HCl~(g)} \textbf{or} conc. \ch{HCl}, ahd. \ch{ZnCl2} cat. (Lucas' reagent); conditions: heat
    \item \ch{ROH + PCl5 -> RCl + POCl3 + HCl}; reagents: ahd. \ch{PCl5}; conditions: r.t.p. Test for alcohol; observations: white fumes of \ch{HCl}
    \item \ch{ROH + SOCl2 -> RCl + SO2 + HCl}; reagents: ahd. \ch{SOCl2}; conditions: r.t.p.
    \item \ch{ROH + HBr -> RBr + H2O}; reagents: \ch{NaBr}, conc. \ch{H2SO4} cat. \textbf{or} \ch{HBr~(g)}; conditions: heat
    \item \ch{RCOOH + PCl5 -> RCOCl + HCl + POCl3}; reagents: ahd. \ch{PCl5}; conditions: r.t.p.
    \item \ch{3 RCOOH + PCl3 -> 3 RCOCl + H3PO3}; reagents: ahd. \ch{PCl3}; conditions: heat
    \item \ch{RCOOH + SOCl2 -> RCOCl + HCl + SO23}; reagents: ahd. \ch{SOCl2}; conditions: r.t.p.
\end{enumerate}
\section{Nucleophilic Addition}
\begin{enumerate}
    \item \ch{RC(=O)R' + HCN -> RC(OH)(CN)R'}; reagents: \ch{HCN}, trace \ch{NaCN} or \ch{NaOH} cat.; conditions: \SI{10}{\celsius} to \SI{20}{\celsius}
\end{enumerate}
\section{Oxidation}
\begin{enumerate}
    \item \ch{CH2=CH2 + [O] + H2O -> CH2OHCH2OH}; reagents, conditions: cold \ch{KMnO4} in \ch{H2SO4}/\ch{NaOH}; observations: decolourisation of purple \ch{KMnO4}, brown ppt of \ch{MnO2}
    \item Strong oxidation of alkenes; reagents, conditions: hot \ch{KMnO4} in \ch{H2SO4}/\ch{NaOH}; observations: decolourisation of purple \ch{KMnO4}; effervescence of \ch{CO2} (if terminal alkene)
    
    \ch{R1C(R2)=C(R3)R4 + 2 [O] -> R1C(=O)R2 + R3C(=O)R4}; \ch{R1C(R2)=CHR3 + 3 [O] -> R1C(=O)R2 + R4COOH}
    
    \ch{R1CH=CHR2 + 4 [O] -> R1COOH + R2COOH}; \ch{R1C(R2)=CH2 + 4 [O] -> R1C(=O)R2 + CO2 + H2O}
    \item \ch{C6H5R + 3 [O] -> C6H5COOH + (CO2)}; reagents: \ch{KMnO4} in \ch{H2SO4~(aq)}; conditions: heat; observations: purple \ch{KMnO4} is decolourised, white ppt of benzoic acid. Note: \ch{R} is any alkyl group with at least one \ch{H} attached to the \ch{C} attached to the ring. \ch{CO2} if \ch{R} contains more than one \ch{C}. Balance accordingly.
    \item \ch{RCH2OH + [O] -> RCHO + H2O}; reagents: \ch{K2Cr2O7~(aq)} in \ch{H2SO4~(aq)}; conditions: heat under distillation; observations: orange \ch{K2Cr2O7} turns green.
    \item \ch{RCH2OH + 2 [O] -> RCOOH + H2O}; reagents: \ch{KMnO4~(aq)} or \ch{K2Cr2O7~(aq)} in \ch{H2SO4~(aq)}; conditions: heat; observations: purple \ch{KMnO4} decolourises or orange \ch{K2Cr2O7} turns green
    \item \ch{RCHOHR' + [O] -> RC(=O)R' + H2O}; reagents: \ch{KMnO4~(aq)} or \ch{K2Cr2O7~(aq)} in \ch{H2SO4~(aq)}; conditions: heat; observations: purple \ch{KMnO4} decolourises or orange \ch{K2Cr2O7} turns green. Test for 1\SIUnitSymbolDegree{} and 2\SIUnitSymbolDegree{} alcohol -- 3\SIUnitSymbolDegree{} alcohol has no colour change.
    \item \ch{RCHO + [O] -> RCOOH}; reagents: \ch{KMnO4~(aq)} or \ch{K2Cr2O7~(aq)} in \ch{H2SO4~(aq)}; conditions: heat; observations: purple \ch{KMnO4} decolourises or orange \ch{K2Cr2O7} turns green
    \item \ch{RCHOHCH3 + 4 I2 + 6 NaOH -> CHI3 + 5 NaI + RCOO^-Na+ + 5 H2O}; reagents: \ch{I2~(aq)} in \ch{NaOH}; conditions: heat; observations: pale yellow ppt
    \item \ch{RC(=O)CH3 + 3 I2 + 4 NaOH -> CHI3 + 3 NaI + RCOO^-Na+ + 3 H2O}; reagents: \ch{I2~(aq)} in \ch{NaOH}; conditions: heat; observations: pale yellow ppt
    
    Note: \ch{R} must be an alkyl or aryl; the \ch{(=O)} group cannot be part of a carboxylic acid, ester, or acyl chloride.
    \item \ch{RCHO + 2 [Ag(NH3)2]+ + 3 OH- -> RCOO- + 2 Ag + 4 NH3 + 2 H2O}; reagents: Tollens' reagent (\ch{NH3~(aq)} with \ch{AgNO3}); conditions: heat; observations: silver mirror or black ppt. Test for all aldehydes.
    \item \ch{RCHO + 2 Cu^2+ + 5 OH- -> RCOO- + Cu2O + 3 H2O}; reagents: Fehling's reagent; conditions: heat; observations: reddish-brown ppt. Test for aliphatic (only) aldehydes.
\end{enumerate}
\section{Reduction}
\begin{enumerate}
    \item \ch{CH2=CH2 + 2 [H] -> CH3CH3}; reagents: \ch{H2~(g)}; conditions: \ch{Ni} cat., h.t.p. \textbf{or} \ch{Pd} cat. or \ch{Pt} cat, r.t.p.; observations: decrease in pressure
    \item \ch{RCHO + 2 [H] -> RCH2OH}; reagents, conditions: \ch{LiAlH4} in dry ether or \ch{NaBH4} in (m)ethanol or \ch{Na} in ethanol or \ch{H2} with \ch{Pd}/\ch{Pt} cat., r.t.p. \textbf{or} \ch{H2} with {Ni} cat., heat
    \item \ch{RC(=O)R' + 2 [H] -> RCHOHR'}; reagents, conditions: \ch{LiAlH4} in dry ether or \ch{NaBH4} in (m)ethanol or \ch{Na} in ethanol or \ch{H2} with \ch{Pd}/\ch{Pt} cat., r.t.p. \textbf{or} \ch{H2} with {Ni} cat., heat
    \item \ch{RCOOH + 4 [H] -> RCH2OH + H2O}; reagents: \ch{LiAlH4} in dry ether; conditions: r.t.p.
    \item \ch{RCOOR' + 4 [H] -> RCH2OH + R'OH}; reagents: \ch{LiAlH4} in dry ether; conditions: r.t.p
    \item \ch{RCONR'R'' + 4 [H] -> RCH2NR'R'' + H2O}; reagents: \ch{LiAlH4} in dry ether; conditions: r.t.p
    \item \ch{RCN + 4 [H] -> RCH2NH2}; reagents, conditions: \ch{LiAlH4} in dry ether or \ch{H2}, \ch{Pd}/\ch{Pt} cat., r.t.p. \textbf{or} \ch{H2}, \ch{Ni} cat., heat. Note: if using \ch{H2}, then use \ch{H2} in equation
    \item \ch{C6H5NO2 + 6 [H] -> C6H5NH2 + 2 H2O}; reagents, conditions: \ch{Sn} in exc. conc. \ch{HCl}, heat then \ch{NaOH~(aq)}, r.t.p.
\end{enumerate}
\section{Elimination}
\begin{enumerate}
    \item \ch{CH3CH2X + KOH -> CH2=CH2 + KX + H2O}; reagents: ethanolic \ch{KOH}; conditions: heat.
    \item \ch{CH3CH2OH -> CH2=CH2 + H2O}; reagents, conditions: exc. conc. \ch{H2SO4} cat, \SI{180}{\celsius} or exc. conc. \ch{H3PO4} cat., \SI{250}{\celsius} or \ch{Al2O3} cat., \SI{350}{\celsius}
\end{enumerate}
\section{Hydrolysis}
\begin{enumerate}
    \item \ch{RCN + 2 H2O + H+ -> RCOOH + NH4+}; reagents: \ch{HCl~(aq)} or \ch{H2SO4~(aq)}; conditions: heat
    \item \ch{RCN + H2O + OH- -> RCOO- + NH3}; reagents: \ch{NaOH~(aq)} or \ch{KOH~(aq)}; conditions: heat
    \item \ch{RCOOR' + H2O <=> RCOOH + R'OH}; reagents: \ch{HCl~(aq)} or \ch{H2SO4~(aq)}; conditions: heat
    \item \ch{RCOOR' + NaOH <=> RCOO^-Na+ + R'OH}; reagents: \ch{NaOH~(aq)} or \ch{KOH~(aq)}; conditions: heat
    \item \ch{RCONR'R'' + H2O <=> RCOOH + N^+H2R'R''}; reagents: \ch{HCl~(aq)} or \ch{H2SO4~(aq)}; conditions: heat
    \item \ch{RCONR'R'' + NaOH -> RCOO^-Na+ + NHR'R''}; reagents: \ch{NaOH~(aq)} or \ch{KOH~(aq)}; conditions: heat
\end{enumerate}
\section{Miscellaneous}
\begin{enumerate}
    \item \ch{ROH + Na -> RO^-Na+ + 1/2 H2}; reagents: \ch{Na} or \ch{K}; conditions: r.t.p.; observations: effervescence of \ch{H2}; ToR: acid-metal displacement
    \item \ch{RCOOH + Na -> RCOO^-Na+ + 1/2 H2}; reagents: \ch{Na} or \ch{K}; conditions: r.t.p.; observations: effervescence of \ch{H2}; ToR: acid-metal displacement
    \item \ch{C6H5OH + NaOH -> C6H5O^-Na+ + H2O}; reagents: \ch{NaOH}; conditions: r.t.p.; ToR: neutralisation
    \item \ch{RCOOH + NaOH -> RCOO^-Na+ + H2O}; reagents: \ch{NaOH}; conditions: r.t.p.; ToR: neutralisation
    \item \ch{2 RCOOH + Na2CO3 -> 2 RCOO^-Na+ + H2O + CO2}; reagents: \ch{Na2CO3}; conditions: r.t.p.; ToR: neutralisation
    \item \ch{RNH2 + HCl -> RNH3^+Cl-}; reagents: mineral acid; conditions: r.t.p.; ToR: neutralisation
    \item \ch{RC(=O)R' + "o,p-" C6H3(NHNH2)(NO2)2 -> "o,p-" C6H3(NHN=C(R)R')(NO2)2 + H2O}; 
    
    reagents: 2,4-dinitrophenylhydrazine (2,4-DNPH i.e. Brady's reagent); conditions: heat; observation: orange crystals of hydrazone; ToR: condensation. Test for all aldehydes and ketones.
    \item \ch{C6H5OH ->[ FeCl3 ] "violet~complex"}; reagents: neutral \ch{FeCl3~(aq)}; conditions: r.t.p.; observations: violet complex formed; ToR: complexation. Test for phenol.
\end{enumerate}
